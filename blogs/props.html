    # 父传子

    父组件通过props向子组件传递数据

        // 父组件
    <ChildComponent name="张三" age="18" />

    // 子组件
    <div>姓名：{this.props.name}</div>
    <div>年龄：{this.props.age}</div>

    # 子传父

    子组件通过props向父组件传递数据

        // 父组件
    <ChildComponent onClick={this.handleClick} />

    // 子组件 
    <button onClick={() => this.props.onClick('张三')}>点击</button>

    # 兄弟组件

    通过共同的父组件传递数据

        // 父组件
    <ChildComponentA />
    <ChildComponentB />

    // 子组件A 
    <button onClick={() => this.props.onClick('张三')}>点击</button>

    // 子组件B
    <button onClick={() => this.props.onClick('李四')}>点击</button>

    // 父组件
    handleClick(name) {
        console.log(name);

        // 子组件B通过props向父组件传递数据
        this.props.onClick(name);

        // 子组件A通过props向父组件传递数据
        this.props.onClick(name);

    }

    # 跨层级组件通信

    通过context实现跨层级组件通信

        // 创建context
    const MyContext = React.createContext();

    // 父组件   
    <MyContext.Provider value={{name: '张三', age: 18}}>
        <ChildComponentA />
        <ChildComponentB />
    </MyContext.Provider>

    // 子组件A
    <MyContext.Consumer >
        {value => <div>姓名：{value.name}</div>}
    </MyContext.Consumer>

    // 子组件B
    <MyContext.Consumer >
        {value => <div>姓名：{value.name}</div>}
    </MyContext.Consumer>

    # 全局状态管理

    通过Redux实现全局状态管理

        // 创建store
    const store = createStore(reducer);

    // 父组件
    <Provider store={store}>
        <ChildComponentA />
        <ChildComponentB />
    </Provider>

    // 子组件A
    <button onClick={() => this.props.onClick('张三')}>点击</button>

    // 子组件B
    <button onClick={() => this.props.onClick('李四')}>点击</button>

    // 子组件A通过props向父组件传递数据
    handleClick(name) {
        console.log(name);

        // 子组件B通过props向父组件传递数据
        this.props.onClick(name);

        // 子组件A通过props向父组件传递数据 
        this.props.onClick(name);   
    }
